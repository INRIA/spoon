[name]
EncloseInBranch

[contract]
a patch should be able to add a branch statement enclosing context

[input]
class A {
    void anchor() {}
    void foo() {}
    
    void m1() {
        anchor();
        foo();
    }
}

[expected]
class A {
    void anchor() {}
    void foo() {}
    
    void m1() {
        boolean debug = Math.random() < 0.5;
        anchor();
        if (debug) {
            foo();
        }
    }
}

[patch]
@@
@@
+ boolean debug = Math.random() < 0.5;
  anchor();
+ if (debug) {
      foo();
+ }
