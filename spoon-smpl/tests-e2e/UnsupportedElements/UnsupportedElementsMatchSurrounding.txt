[name]
UnsupportedElementsMatchSurrounding

[contract]
should be able to match and transform things surrounding an unsupported element

[patch]
@@
@@
- a();
...
b();


[input]
class A {
    void a() {}
    void b() {}
    boolean loopsNotSupported;
    
    void foo() {
        a();
        while (loopsNotSupported) {
          break;
        }
        b();
    }
}


[expected]
class A {
    void a() {}
    void b() {}
    boolean loopsNotSupported;
    
    void foo() {
        while (loopsNotSupported) {
            break;
        } 
        b();
    }
}
