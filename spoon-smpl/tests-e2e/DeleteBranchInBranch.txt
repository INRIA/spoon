[name]
DeleteBranchInBranch

[contract]
a patch should be able to delete a complete branch statement nested inside another branch

[input]
class A {
    void m1() {
        if (somevariable) {
            before();
            
            if (true) {
                int x = 0;
            }
            
            after();
        }
    }
}

[expected]
class A {
    void m1() {
        if (somevariable) {
            before();
            after();
        }
    }
}

[patch]
@@
@@
- if (true) {
-     int x = 0;
- }
